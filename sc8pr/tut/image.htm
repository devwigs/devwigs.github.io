The sc8pr.image module provides an Image class that you can use to load, create, modify, and save graphic images. You can then use your images as backgrounds or sprite costumes, or you can draw them as extra content (for example, a player status or score) when writing your custom drawing function.

Below is a program that creates its own background image programmatically rather than simply loading a saved image. See the reference for more details.

from sc8pr.sketch import Sketch
from sc8pr.image import Image
from sc8pr.util import rgba, SOUTH, NORTH

BLUE, SKYBLUE, YELLOW, WHITE, RED = rgba("blue", "skyblue", "yellow", "white", "red")

def postcard(sk):
    # Draw the sky as the background
    bg = Image(sk.size, SKYBLUE)
    w, h = bg.size

    # Draw the sun
    sun = Image.ellipse(0.12*w, YELLOW)
    bg.blit(sun, bg.center, SOUTH)

    # Draw a sail
    y = 0.65 * h
    sail = (0.1*w, y), (0.12*w, y), (0.105*w, 0.9*y)
    bg.plot(sail, fill=WHITE, stroke=None)

    # Draw the sea
    sea = Image.rect((w, h-y+1), BLUE)
    bg.blit(sea, (0, y))

    # Add some text
    font = sk.loadFont("cursive", h//5)   # Size must be an int!
    text = Image.text("Summer Vacation!", font, RED)
    bg.blit(text, (w/2, 0.1*h), NORTH)
    return bg

def setup(sk):
    sk.setBackground(postcard(sk))

# Run the sketch
Sketch(setup).play()
